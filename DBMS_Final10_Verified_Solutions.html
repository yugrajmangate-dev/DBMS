-- Q1: PROCEDURE — Assign Commission
CREATE OR REPLACE PROCEDURE assign_commission IS
  v_exp NUMBER;
BEGIN
  FOR r IN (SELECT emp_id, salary, DoJ FROM Employee) LOOP
    v_exp := MONTHS_BETWEEN(SYSDATE, r.DoJ) / 12;
    IF    r.salary > 10000 THEN
      UPDATE Employee SET commission = r.salary * 0.004  WHERE emp_id = r.emp_id;
    ELSIF r.salary < 10000 AND v_exp > 10 THEN
      UPDATE Employee SET commission = r.salary * 0.0035 WHERE emp_id = r.emp_id;
    ELSIF r.salary < 3000 THEN
      UPDATE Employee SET commission = r.salary * 0.0025 WHERE emp_id = r.emp_id;
    ELSE
      UPDATE Employee SET commission = r.salary * 0.0015 WHERE emp_id = r.emp_id;
    END IF;
  END LOOP;
  COMMIT;
  DBMS_OUTPUT.PUT_LINE('Commission assigned!');
END;

-- UNNAMED PL/SQL BLOCK
DECLARE
  p_id  NUMBER := &emp_id;
  v_sal NUMBER;
  v_doj DATE;
  v_exp NUMBER;
  v_new NUMBER;
BEGIN
  SELECT salary, DoJ INTO v_sal, v_doj
  FROM Employee WHERE emp_id = p_id;

  v_exp := MONTHS_BETWEEN(SYSDATE, v_doj) / 12;

  IF    v_exp > 10 THEN v_new := v_sal * 1.20;   -- 20% hike
  ELSIF v_exp > 5  THEN v_new := v_sal * 1.10;   -- 10% hike
  ELSE                  v_new := v_sal * 1.05;   -- 5% hike
  END IF;

  UPDATE Employee SET salary = v_new WHERE emp_id = p_id;
  INSERT INTO Salary_Increment VALUES(p_id, v_new);
  COMMIT;

  DBMS_OUTPUT.PUT_LINE('Experience : ' || ROUND(v_exp,1) || ' years');
  DBMS_OUTPUT.PUT_LINE('New Salary : ' || v_new);

EXCEPTION
  WHEN NO_DATA_FOUND THEN
    DBMS_OUTPUT.PUT_LINE('ERROR: Employee ID not found!');
  WHEN OTHERS THEN
    DBMS_OUTPUT.PUT_LINE('ERROR: ' || SQLERRM);
END;
/
