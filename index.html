PS 01 Triggers — Salary Backup & Prevent CEO Delete Oracle PL/SQL

-- TABLE SETUP 
CREATE TABLE Employee (
  emp_id      NUMBER PRIMARY KEY,
  emp_name    VARCHAR2(50),
  salary      NUMBER,
  designation VARCHAR2(30)
);

CREATE TABLE Salary_Backup (
  emp_id           NUMBER,
  old_salary       NUMBER,
  new_salary       NUMBER,
  salary_difference NUMBER
);

INSERT INTO Employee VALUES(1,'Alice',50000,'Manager');
INSERT INTO Employee VALUES(2,'Bob',100000,'CEO');
COMMIT;
-- TRIGGER 1: Record Salary Change 
CREATE OR REPLACE TRIGGER trg_salary_backup
AFTER UPDATE OF salary ON Employee
FOR EACH ROW
BEGIN
  INSERT INTO Salary_Backup VALUES(
    :OLD.emp_id,
    :OLD.salary,
    :NEW.salary,
    :NEW.salary - :OLD.salary
  );
END;
/

-- Test

-- TRIGGER 2: Prevent Deleting CEO 
CREATE OR REPLACE TRIGGER trg_no_delete_ceo
BEFORE DELETE ON Employee
FOR EACH ROW
BEGIN
  IF :OLD.designation = 'CEO' THEN
    RAISE_APPLICATION_ERROR(-20001, 'Cannot delete CEO record!');
  END IF;
END;

-- Q1: PROCEDURE — Simple Interest 
CREATE OR REPLACE PROCEDURE calc_interest(p_acc IN NUMBER, p_rate IN NUMBER) IS
  v_bal     NUMBER;
  v_years   NUMBER;
  v_interest NUMBER;
BEGIN
  SELECT Balance, NoOfYears INTO v_bal, v_years
  FROM Account WHERE Account_No = p_acc;

  v_interest := (v_bal * p_rate * v_years) / 100;

  INSERT INTO Earned_Interest VALUES(p_acc, v_interest);
  COMMIT;

  DBMS_OUTPUT.PUT_LINE('--- Earned Interest Table ---');
  FOR r IN (SELECT * FROM Earned_Interest) LOOP
    DBMS_OUTPUT.PUT_LINE('Acc: '||r.Account_No||' | Interest: '||r.Interest_Amt);
  END LOOP;
END;
/

-- Execute
SET SERVEROUTPUT ON;
EXEC calc_interest(101, 8);
-- Q2: FUNCTION — Accounts with Balance > 50000 
CREATE OR REPLACE FUNCTION high_balance_accounts RETURN SYS_REFCURSOR IS
  c SYS_REFCURSOR;
BEGIN
  OPEN c FOR SELECT * FROM Account WHERE Balance > 50000;
  RETURN c;
END;
/

-- Execute
DECLARE
  c SYS_REFCURSOR;
  r Account%ROWTYPE;
BEGIN
  c := high_balance_accounts();
  LOOP
    FETCH c INTO r;
    EXIT WHEN c%NOTFOUND;
    DBMS_OUTPUT.PUT_LINE(r.Account_No||' | '||r.Cust_Name||' | '||r.Balance);
  END LOOP;
  CLOSE c;
END;

PS 12 PL/SQL — Commission Procedure & Manager Function Oracle PL/SQL

-- Q1: PROCEDURE — Assign Commission 
CREATE OR REPLACE PROCEDURE assign_commission IS
  v_exp NUMBER;
BEGIN
  FOR r IN (SELECT emp_id, salary, DoJ FROM Employee) LOOP
    v_exp := MONTHS_BETWEEN(SYSDATE, r.DoJ) / 12;
    IF r.salary > 10000 THEN
      UPDATE Employee SET commission = r.salary * 0.004 WHERE emp_id = r.emp_id;
    ELSIF v_exp > 10 THEN
      UPDATE Employee SET commission = r.salary * 0.0035 WHERE emp_id = r.emp_id;
    ELSIF r.salary < 3000 THEN
      UPDATE Employee SET commission = r.salary * 0.0025 WHERE emp_id = r.emp_id;
    ELSE
      UPDATE Employee SET commission = r.salary * 0.0015 WHERE emp_id = r.emp_id;
    END IF;
  END LOOP;
  COMMIT;
  DBMS_OUTPUT.PUT_LINE('Commission assigned successfully!');
END;
/

EXEC assign_commission;

CREATE OR REPLACE FUNCTION get_manager_name(p_dept_id IN NUMBER) RETURN VARCHAR2 IS

-- UNNAMED BLOCK with Exception Handling 
DECLARE
  p_id      NUMBER := &emp_id;
  v_sal     NUMBER;
  v_doj     DATE;
  v_exp     NUMBER;
  v_new_sal NUMBER;
  e_not_found EXCEPTION;
BEGIN
  SELECT salary, DoJ INTO v_sal, v_doj
  FROM Employee WHERE emp_id = p_id;

  v_exp := MONTHS_BETWEEN(SYSDATE, v_doj) / 12;

  IF v_exp > 10 THEN
    v_new_sal := v_sal * 1.20;    -- 20% hike
  ELSIF v_exp > 5 THEN
    v_new_sal := v_sal * 1.10;    -- 10% hike
  ELSE
    v_new_sal := v_sal * 1.05;    -- 5% hike
  END IF;

  UPDATE Employee SET salary = v_new_sal WHERE emp_id = p_id;
  INSERT INTO Salary_Increment VALUES(p_id, v_new_sal);
  COMMIT;

  DBMS_OUTPUT.PUT_LINE('Experience: ' || ROUND(v_exp,1) || ' years');
  DBMS_OUTPUT.PUT_LINE('New Salary: ' || v_new_sal);

EXCEPTION
  WHEN NO_DATA_FOUND THEN
    DBMS_OUTPUT.PUT_LINE('Error: Employee ID not found!');
  WHEN OTHERS THEN
    DBMS_OUTPUT.PUT_LINE('Error: ' || SQLERRM);
END;
